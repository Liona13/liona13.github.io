Example Code


import numpy as np
import matplotlib.pyplot as plt

# Create data for visualization
x = np.linspace(-10, 10, 100)
y = x**2

# Create plot
plt.plot(x, y)
plt.title('Quadratic Function')
plt.xlabel('x')
plt.ylabel('y')
plt.grid(True)
plt.show()
